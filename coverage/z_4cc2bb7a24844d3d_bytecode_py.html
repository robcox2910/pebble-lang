1"""Bytecode representation for the Pebble compiler.

140@dataclass 

141class CodeObject: 

142 """Instruction sequence and constant pool for one compilation unit. 

143 

144 Use :meth:`add_constant` to populate the constant pool — it deduplicates 

145 by both value *and* type so that ``0`` (int) and ``False`` (bool) remain 

146 distinct entries. 

147 

148 Attributes: 

149 name: Human-readable label (``"<main>"`` or the function name). 

150 instructions: Mutable list built up during compilation. 

151 constants: Deduplicated pool of literal values. 

152 parameters: Ordered list of parameter names for function code objects. 

153 

154 """ 

155 

156 name: str 

157 instructions: list[Instruction] = field(default_factory=lambda: []) 

158 constants: list[int | float | str | bool | None] = field(default_factory=lambda: []) 

159 parameters: list[str] = field(default_factory=lambda: []) 

160 cell_variables: list[str] = field(default_factory=lambda: []) 

161 free_variables: list[str] = field(default_factory=lambda: []) 

162 param_types: list[TypeAnnotation | None] = field(default_factory=lambda: []) 

163 return_type: TypeAnnotation | None = None 

164 is_generator: bool = False 

165 is_async: bool = False 

166 

167 _constant_index: dict[tuple[type, int | float | str | bool | None], int] = field( 

168 default_factory=lambda: {}, 

169 repr=False, 

170 ) 

171 

172 def add_constant(self, value: int | float | str | bool | None) -> int: 

173 """Add *value* to the constant pool and return its index. 

174 

175 Duplicate entries (same value *and* type) are reused. A cache 

176 dictionary keeps lookups O(1) instead of scanning the pool. 

177 """ 

178 key = (type(value), value) 

179 if key in self._constant_index: 

180 return self._constant_index[key] 

181 idx = len(self.constants) 

182 self.constants.append(value) 

183 self._constant_index[key] = idx 

184 return idx
